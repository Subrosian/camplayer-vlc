--- Initial Setup
sudo apt update
sudo apt install git
sudo apt install golang
sudo apt install vlc
git clone https://github.com/Subrosian/camplayer-vlc
cd camplayer-vlc/
go build -o camplayer-vlc .
sudo cp camplayer-vlc /usr/local/bin/
sudo chown root:root /usr/local/bin/camplayer-vlc
sudo chmod 755 /usr/local/bin/camplayer-vlc

--- Update Project
git pull
go build -o camplayer-vlc .
sudo cp camplayer-vlc /usr/local/bin/
sudo chown root:root /usr/local/bin/camplayer-vlc
sudo chmod 755 /usr/local/bin/camplayer-vlc

--- Make config
sudo nano /etc/camplayer-vlc.conf
***
RTSP_URL=rtsp://user:pass@camera-ip:554/stream

--- Make service
sudo nano /etc/systemd/system/camplayer-vlc.service
***
[Unit]
Description=camplayer-vlc RTSP player
After=network-online.target
Wants=network-online.target

[Service]
Type=simple

User=pi
Group=pi
WorkingDirectory=/home/pi
Environment=HOME=/home/pi

ExecStart=/usr/local/bin/camplayer-vlc

Restart=always
RestartSec=5

StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target
