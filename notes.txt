--- Initial Setup
sudo apt update
sudo apt install git
sudo apt install golang
sudo apt install vlc
git clone https://github.com/Subrosian/camplayer-vlc
cd camplayer-vlc/
go build -o camplayer-vlc .

--- Update Project
git pull
go build -o camplayer-vlc .
